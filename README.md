色々インストールする:

```
$ npm install
```

gulpを立ち上げる:

```
$ gulp
```

config.tomlの例:

```
baseURL = "http://example.org/"
languageCode = "ja"
title = "Test"
theme = "hugo-bootstrap"
copyright = "copyright"

[params]
description = "description"
keywords = "keywords"
googleAnalyticsId = ""
fbAppId = ""

[permalinks]
post = "/:year/:month/:title/"

[taxonomies]
tag = "tags"
category = "categories"
```
